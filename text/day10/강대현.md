def enque(item):
    global que, rear
    rear += 1
    que[rear]=item

def deque():
    global front
    front += 1
    temp = que[front]
    que[front] = 0
    return temp

def bfs(start):
    enque(start)

    while rear != front :
        now = deque()

                visited[nxt] = visited[now] + 1
                enque(nxt)
