def sol(start_y,start_x,k):
    global result
    for i in range(2):
        visited = [[0] * n for _ in range(n)]
        visited[start_y][start_x] = 1
        count = 1
        stack = [(start_y,start_x)]
        while stack:
            y,x = stack.pop(0)
            for j in range(2):
                ny = y + dy[i][j]
                nx = x + dx[i][j]
                if 0<=nx<n and 0<=ny<n and data[ny][nx] and not visited[ny][nx]:
                    visited[ny][nx] = 1
                    stack.append((ny,nx))
                    count+=1
        if count == k:
            result += 1

MIS = lambda: map(int,input().split())
